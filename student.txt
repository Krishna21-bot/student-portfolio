        function showContent(type) {

            if (type === 'achievements') {
                contentDiv.innerHTML = "<h2>Achievements</h2><p>Won XYZ coding competition, Published research paper...</p>";

                contentDiv.innerHTML = "<h2>Skills</h2><p>HTML, CSS, JavaScript, PHP, Python...</p>";
